<script>

    async function sendMessage() {

        let userMessage = document.getElementById("userInput").value;

        let apiKey = document.getElementById("apiKey").value;  // Optional, if you want extra auth


        if (!apiKey) {

            alert("Please enter your API key.");

            return;

        }


        document.getElementById("chatbox").innerHTML += `<p><strong>You:</strong> ${userMessage}</p>`;


        let response = await fetch("https://xyz123.execute-api.us-east-1.amazonaws.com", {

            method: "POST",

            headers: { "Content-Type": "application/json" },

            body: JSON.stringify({ message: userMessage })

        });


        let data = await response.json();


        if (data.error) {

            document.getElementById("chatbox").innerHTML += `<p><strong>Error:</strong> ${data.error}</p>`;

        } else {

            document.getElementById("chatbox").innerHTML += `<p><strong>GPT:</strong> ${data.response}</p>`;

        }


        document.getElementById("userInput").value = "";

    }

</script>
